import  { %%PascalCase%% } from "./%%kebap-case%%-element";

let %%PascalCase%%;

describe("Test Case for the %%PascalCase%% Class", () => {
    beforeEach(() => {
        %%PascalCase%% = new %%PascalCase%%();
    })

    afterEach(() => {
        %%PascalCase%% = null;
    });
})


test ("%%PascalCase%% has a property 'name' with value of '%%PascalCase%%'", () => {
    const %%PascalCase%% = new %%PascalCase%%();
    expect(%%PascalCase%%.name).toBe('%%PascalCase%%');
});
